        const prompt = `Based on the event "${market.question}",

### –í–∞—Ä–∏–∞–Ω—Ç 2: –ö–æ–º–±–∏–Ω–∏—Ä–æ–≤–∞–Ω–Ω—ã–π –ø–æ–¥—Ö–æ–¥

    // –®–∞–≥ 1: –§–∏–ª—å—Ç—Ä–∞—Ü–∏—è –ø–æ –∫–∞—á–µ—Å—Ç–≤—É (–Ω–∞—à –ø–æ–¥—Ö–æ–¥)

        .map(s => s.market);
    
    // –®–∞–≥ 2: –ü–æ–∏—Å–∫ edge (–ø–æ–¥—Ö–æ–¥ Poly-Trader)
    const edges = await Promise.all(
        qualityFiltered.map(async (market) => {
            const aiPrediction = await this.aiPrediction.predictOutcome(market);
            const marketPrice = market.tokens[0].price;
            const edge = (aiPrediction / 100) - marketPrice;

    // –®–∞–≥ 3: –§–∏–ª—å—Ç—Ä—É–µ–º –ø–æ edge

        .filter(e => e.edge >= 0.10) // –ú–∏–Ω–∏–º—É–º 10% edge
        .sort((a, b) => b.edge - a.edge)
        .slice(0, 5)
        .map(e => e.market);
